\begin{itemize}
\item $ x_{a} \prec x_{b} $ (dominate)
\begin{eqnarray*}
\forall k, & f_{k} (x_{a}) \leq f_{k} (x_{b}) \\
\exists k, & f_{k} (x_{a}) < f_{k} (x_{b})
\end{eqnarray*}
\item non-dominance
\begin{eqnarray*}
\nexists x \in \mathbb{X}, x \prec x^{*}
\end{eqnarray*}
\end{itemize}
